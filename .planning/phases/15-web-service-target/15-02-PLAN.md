### Step 6 — Generate app.py

```bash
export DEPLOY_DIR MODEL_NAME VERSION PROBLEM_TYPE
python3 << 'PYEOF'
import json, os

deploy_dir   = os.environ['DEPLOY_DIR']
model_name   = os.environ['MODEL_NAME']
version      = os.environ['VERSION']
problem_type = os.environ['PROBLEM_TYPE']

with open(os.path.join(deploy_dir, 'deployment_metadata.json')) as f:
    meta = json.load(f)

app_py = f'''#!/usr/bin/env python3
"""
DoML web service — generated by /doml-deploy-web
Model: {model_name}
Deployment: {version}
"""
import json
from pathlib import Path
from typing import Any

import joblib
import pandas as pd
from fastapi import FastAPI, Request
from fastapi.responses import HTMLResponse, JSONResponse
from fastapi.templating import Jinja2Templates
from pydantic import create_model

# ── Bootstrap ────────────────────────────────────────────────────────────────
_HERE = Path(__file__).parent
with open(_HERE / "deployment_metadata.json") as _f:
    _metadata = json.load(_f)

MODEL_NAME     = _metadata["model_name"]
PROBLEM_TYPE   = _metadata.get("problem_type", {json.dumps(problem_type)})
FEATURE_SCHEMA = _metadata["feature_schema"]
VERSION        = _metadata.get("version", {json.dumps(version)})

# Dynamic Pydantic request model (D-02)
_TYPE_MAP = {{"int64": float, "float64": float, "object": str}}
_fields = {{
    feat["name"]: (_TYPE_MAP.get(feat["type"], str), ...)
    for feat in FEATURE_SCHEMA
}}
PredictRequest = create_model("PredictRequest", **_fields)

# Load model
_model = joblib.load(_HERE / "model.pkl")

# Jinja2 templates
_templates = Jinja2Templates(directory=str(_HERE / "templates"))

# ── App ───────────────────────────────────────────────────────────────────────
app = FastAPI(title=MODEL_NAME, version=VERSION)


@app.get("/health")
async def health() -> dict:
    return {{"status": "ok", "model_name": MODEL_NAME, "version": VERSION}}


@app.get("/schema")
async def schema() -> dict:
    return {{"feature_schema": FEATURE_SCHEMA}}


@app.get("/", response_class=HTMLResponse)
async def form(request: Request):
    return _templates.TemplateResponse("index.html", {{
        "request":      request,
        "model_name":   MODEL_NAME,
        "features":     FEATURE_SCHEMA,
        "problem_type": PROBLEM_TYPE,
    }})


@app.post("/predict")
async def predict(body: PredictRequest) -> JSONResponse:
    df  = pd.DataFrame([body.model_dump()])
    raw = _model.predict(df)[0]
    pred_val = raw.item() if hasattr(raw, "item") else raw
    result: dict[str, Any] = {{"prediction": pred_val}}

    if PROBLEM_TYPE in ("classification", "binary_classification", "multiclass_classification"):
        if hasattr(_model, "predict_proba"):
            proba   = _model.predict_proba(df)[0]
            classes = _model.classes_
            result["probabilities"] = {{
                str(cls): float(p) for cls, p in zip(classes, proba)
            }}

    return JSONResponse(result)
'''

out_path = os.path.join(deploy_dir, 'app.py')
with open(out_path, 'w') as f:
    f.write(app_py)
print(f'Generated: {out_path}')
PYEOF
if [ $? -ne 0 ]; then
  echo "Failed to generate app.py"
  exit 1
fi
```

              <tbody>${rows}</tbody>
            </table>`;
          }
          html += `</div>`;
          resultDiv.innerHTML = html;
        }

      } finally {
        btn.disabled = false;
      }
    });
  </script>
</body>
</html>'''

out_path = os.path.join(templates_dir, 'index.html')
