## ORT version pin:
- Package: onnxruntime-web@1.17.3
- `ort.min.js`: downloaded from unpkg CDN at generation time (~480 KB text)
- `ort-wasm.wasm`: downloaded from unpkg CDN at generation time (~2.5 MB binary)
- CDN URLs:
  - `https://unpkg.com/onnxruntime-web@1.17.3/dist/ort.min.js`
  - `https://unpkg.com/onnxruntime-web@1.17.3/dist/ort-wasm.wasm`
- ORT files cached at: `/tmp/doml_ort_1.17.3/` to avoid re-downloading on repeated runs

## skl2onnx conversion approach (D-02, D-03):
- Runs inside Docker: `docker compose exec jupyter python3 /tmp/doml_convert_onnx.py`
- The conversion script:
  1. Loads `models/best_model.pkl` (full sklearn Pipeline)
  2. Reads `deployment_metadata.json` for feature_schema and categories
  3. Builds `category_map: {feature_name: {label: integer_code}}` from categories lists
  4. Replaces any OneHotEncoder in the Pipeline's ColumnTransformer with a pre-fitted OrdinalEncoder using categories from the map (D-03)
  5. Calls `skl2onnx.convert_sklearn(onnx_pipeline, initial_types=[('float_input', FloatTensorType([None, n_features]))])`
  6. Writes the ONNX protobuf to `src/<modelname>/vN/model.onnx` (container path)
  7. Prints ONNX output names and input names to stdout for embedding in HTML

## index.html structure (WASM-01, WASM-02, WASM-03):
```html
<!DOCTYPE html>
<html>
<head>
  <meta charset="utf-8">
  <title>DoML — {model_name} Inference</title>
  <style>/* minimal inline styles — see Specifics in CONTEXT.md */</style>
</head>
<body>
  <h1>{model_name}</h1>
  <p>Version: {version} | Type: {problem_type} | Fields: {n_features}</p>

  <!-- Auto-generated form (WASM-02) -->
  <form id="predict-form">
    <!-- Inputs built by JS from FEATURE_SCHEMA at DOMContentLoaded -->
  </form>
  <div id="result" hidden></div>

  <!-- ORT JS inlined directly (not base64 URI — plain text embed) -->
  <script>
  {ort_js_text}
  </script>

  <!-- Inference harness -->
  <script>
  const ORT_WASM_B64 = "{wasm_b64}";   // base64 of ort-wasm.wasm
  const MODEL_B64    = "{model_b64}";   // base64 of model.onnx
  const FEATURE_SCHEMA = {feature_schema_json};
  const CATEGORIES     = {category_map_json};   // {feature_name: {label: int}}
  const PROBLEM_TYPE   = "{problem_type}";

  (async () => {{
    // 1. Point ORT at the embedded WASM (D-01)
    const wasmBytes = Uint8Array.from(atob(ORT_WASM_B64), c => c.charCodeAt(0));
    const wasmBlob  = new Blob([wasmBytes], {{type: 'application/wasm'}});
    const wasmUrl   = URL.createObjectURL(wasmBlob);
    ort.env.wasm.numThreads = 1;  // disable multi-threading for offline use
    ort.env.wasm.wasmPaths  = {{
      'ort-wasm.wasm':               wasmUrl,
      'ort-wasm-simd.wasm':          wasmUrl,
      'ort-wasm-threaded.wasm':      wasmUrl,
      'ort-wasm-simd-threaded.wasm': wasmUrl
    }};

    // 2. Load ONNX model from embedded base64
    const modelBytes   = Uint8Array.from(atob(MODEL_B64), c => c.charCodeAt(0));
    const session      = await ort.InferenceSession.create(modelBytes);
    const inputName    = session.inputNames[0];
    const outputNames  = session.outputNames;

    // 3. Build prediction form from feature_schema (WASM-02)
    const form = document.getElementById('predict-form');
    for (const feat of FEATURE_SCHEMA) {{
      const label = document.createElement('label');
      label.textContent = feat.name + ': ';
      if (feat.categories) {{
        const sel = document.createElement('select');
        sel.name = feat.name;
        for (const cat of feat.categories) {{
          const opt = document.createElement('option');
          opt.value = cat;
          opt.textContent = cat;
          sel.appendChild(opt);
        }}
        label.appendChild(sel);
      }} else {{
        const inp = document.createElement('input');
        inp.type = 'number';
        inp.name = feat.name;
        inp.step = 'any';
        inp.placeholder = feat.example || '0';
        inp.required = true;
        label.appendChild(inp);
      }}
      const row = document.createElement('div');
      row.appendChild(label);
      form.appendChild(row);
    }}
    const btn = document.createElement('button');
    btn.type = 'submit';
    btn.textContent = 'Predict';
    form.appendChild(btn);

    // 4. Handle form submit — run ONNX inference (WASM-03)
    form.addEventListener('submit', async (e) => {{
      e.preventDefault();
      const fd = new FormData(form);
      const data = new Float32Array(FEATURE_SCHEMA.length);
      for (let i = 0; i < FEATURE_SCHEMA.length; i++) {{
        const feat = FEATURE_SCHEMA[i];
        const val = fd.get(feat.name);
        data[i] = feat.categories ? CATEGORIES[feat.name][val] : parseFloat(val);
      }}
      const tensor = new ort.Tensor('float32', data, [1, FEATURE_SCHEMA.length]);
      const results = await session.run({{[inputName]: tensor}});

      const resultDiv = document.getElementById('result');
      resultDiv.removeAttribute('hidden');
      // Display result — regression: numeric; classification: label + optional probs
      if (PROBLEM_TYPE === 'regression') {{
        const pred = results[outputNames[0]].data[0];
        resultDiv.innerHTML = '<strong>Prediction:</strong> ' + pred.toFixed(4);
      }} else {{
        // Classification: label in first output, probability map in second (ZipMap)
        const labelOutput = outputNames.find(n => n.includes('label')) || outputNames[0];
        const label = results[labelOutput].data[0];
        let html = '<strong>Predicted class:</strong> ' + label;
        const probOutput = outputNames.find(n => n.includes('prob'));
        if (probOutput) {{
          const probMap = results[probOutput].data[0];  // JS object: {{class: prob}}
          const entries = Object.entries(probMap).sort((a,b) => b[1]-a[1]);
          html += '<br><strong>Probabilities:</strong><br>';
          for (const [cls, prob] of entries) {{
            const pct = (prob * 100).toFixed(1);
            html += cls + ': ' + pct + '%<br>';
          }}
        }}
        resultDiv.innerHTML = html;
      }}
    }});
  }})().catch(err => {{
    document.getElementById('result').removeAttribute('hidden');
    document.getElementById('result').innerHTML =
      '<span style="color:red">Error: ' + err.message + '</span>';
  }});
  </script>
</body>
</html>
```

if [ ! -f "$ORT_WASM" ]; then
  echo "Downloading ort-wasm.wasm (onnxruntime-web ${ORT_VERSION})..."
  curl -fsSL "https://unpkg.com/onnxruntime-web@${ORT_VERSION}/dist/ort-wasm.wasm" -o "$ORT_WASM"
  if [ $? -ne 0 ]; then
    echo "Failed to download ort-wasm.wasm. Check internet connection."
    echo "Alternatively, place the file at: $ORT_WASM"

  fi

**Step 8 — Size gate check (WASM-05)**

```bash
python3 << PYEOF
import base64, sys, os

ort_js_path   = '${ORT_JS}'
wasm_path     = '${ORT_WASM}'
onnx_path     = '${DEPLOY_DIR}/model.onnx'

ort_js_bytes  = open(ort_js_path,  'rb').read()
wasm_bytes    = open(wasm_path,    'rb').read()
onnx_bytes    = open(onnx_path,   'rb').read()

total_bytes = (
    len(ort_js_bytes) +                    # ORT JS inline text (bytes)
    len(base64.b64encode(wasm_bytes)) +    # WASM base64 string length
    len(base64.b64encode(onnx_bytes))      # model.onnx base64 string length
)
limit = 20 * 1024 * 1024  # 20 MB

ort_mb  = len(ort_js_bytes)  / (1024*1024)
wasm_mb = len(wasm_bytes)    / (1024*1024)
onnx_mb = len(onnx_bytes)    / (1024*1024)
total_mb = total_bytes       / (1024*1024)

print(f"Bundle sizes:")
print(f"  ort.min.js:   {ort_mb:.1f} MB")
print(f"  ort-wasm:     {wasm_mb:.1f} MB (base64)")
print(f"  model.onnx:   {onnx_mb:.1f} MB (base64)")
print(f"  Total:        {total_mb:.1f} MB")

if total_bytes > limit:
    print(f"")
    print(f"ONNX/WASM bundle exceeds 20 MB ({total_mb:.1f} MB).")
    print(f"The model.onnx ({onnx_mb:.1f} MB) is too large to embed in a browser page.")
    print(f"")
    print(f"Consider the web service target: /doml-deploy-web")
    sys.exit(1)
else:
    print(f"Size gate passed: {total_mb:.1f} MB < 20 MB")
PYEOF
if [ $? -ne 0 ]; then exit 1; fi
```
