**Cell 5 — Target-specific benchmark helpers (Python code cell):**

Write all three helper function sets in one cell guarded by `if target ==` blocks:

```python
# ── CLI BINARY HELPERS ──────────────────────────────────────────────────────
if target == 'cli_binary':
    binary_path = str(deploy_dir / 'dist' / 'predict' / 'predict')

    def run_cli_single(row_dict):
        """Invoke CLI binary with JSON input; return prediction."""
        result = subprocess.run(
            [binary_path, '--input', json.dumps(row_dict)],
            capture_output=True, text=True, timeout=30
        )
        if result.returncode != 0:
            raise RuntimeError(f"CLI binary failed (exit {result.returncode}): {result.stderr.strip()}")
        return json.loads(result.stdout.strip())

    run_single_predict = run_cli_single
    print(f"CLI binary: {binary_path}")

# ── WEB SERVICE HELPERS ──────────────────────────────────────────────────────
elif target == 'web_service':
    compose_file = str(deploy_dir / 'docker-compose.serve.yml')
    SERVICE_URL  = 'http://localhost:8080'

    def start_web_service():
        subprocess.run(
            ['docker', 'compose', '-f', compose_file, 'up', '-d'],
            check=True, capture_output=True, text=True
        )
        deadline = time.time() + 30
        while time.time() < deadline:
            try:
                r = requests.get(f'{SERVICE_URL}/health', timeout=2)
                if r.status_code == 200:
                    print(f"Web service healthy: {r.json()}")
                    return
            except (requests.ConnectionError, requests.Timeout):
                pass
            time.sleep(1)
        raise TimeoutError("Web service did not start within 30 seconds")

    def stop_web_service():
        subprocess.run(
            ['docker', 'compose', '-f', compose_file, 'down'],
            check=False, capture_output=True, text=True
        )

    def run_web_single(row_dict):
        """POST /predict with JSON; return prediction value."""
        r = requests.post(f'{SERVICE_URL}/predict', json=row_dict, timeout=10)
        r.raise_for_status()
        resp = r.json()
        return resp.get('prediction', resp)

    run_single_predict = run_web_single
    print(f"Web service compose: {compose_file}")

# ── ONNX/WASM HELPERS (via Python onnxruntime — approximation) ───────────────
elif target == 'onnx_wasm':
    import onnxruntime as ort
    onnx_path = str(deploy_dir / 'model.onnx')
    ort_session = ort.InferenceSession(onnx_path, providers=['CPUExecutionProvider'])
    ort_input_name = ort_session.get_inputs()[0].name
    print(f"ONNX model: {onnx_path}")
    print(f"ORT input name: {ort_input_name}")
    print("NOTE: Benchmarked via Python onnxruntime; browser (WASM) performance may vary")

    def run_onnx_single(row_dict):
        """Build float32 tensor from row_dict using feature_schema ordinal encoding."""
        values = []
        for feat in feature_schema:
            name = feat['name']
            if name not in row_dict:
                continue
            val = row_dict[name]
            if feat.get('categories') and name in category_map:
                val = category_map[name].get(str(val), 0)
            values.append(float(val))
        tensor = np.array([values], dtype=np.float32)
        result = ort_session.run(None, {ort_input_name: tensor})
        return result[0][0]

    run_single_predict = run_onnx_single

else:
    raise ValueError(f"Unknown deployment target: '{target}'. Expected cli_binary, web_service, or onnx_wasm.")
```

**Cell 6 — Single-prediction latency benchmark (Python code cell, PERF-02):**

```python
# PERF-02: Single-prediction latency — 5 warm-up runs discarded, 1000 timed runs
N_RUNS = 1000
N_WARMUP = 5
sample_row = X_test.iloc[0].to_dict()

# For web service: start container before timing, stop in finally
_web_started = False
try:
    if target == 'web_service':
        start_web_service()
        _web_started = True

    # Warm-up
    for _ in range(N_WARMUP):
        run_single_predict(sample_row)

    # Timed runs
    latencies_s = []
    for _ in range(N_RUNS):
        t0 = time.perf_counter()
        run_single_predict(sample_row)
        latencies_s.append(time.perf_counter() - t0)

    mean_ms  = float(np.mean(latencies_s)) * 1000
    std_ms   = float(np.std(latencies_s))  * 1000
    p50_ms   = float(np.percentile(latencies_s, 50)) * 1000
    p95_ms   = float(np.percentile(latencies_s, 95)) * 1000
    p99_ms   = float(np.percentile(latencies_s, 99)) * 1000

    display(Markdown(f"""
### Single-Prediction Latency (n={N_RUNS})

| Metric | Value |
|--------|-------|
| Mean   | {mean_ms:.2f} ms |
| Std    | {std_ms:.2f} ms |
| p50    | {p50_ms:.2f} ms |
| p95    | {p95_ms:.2f} ms |
| p99    | {p99_ms:.2f} ms |
"""))
except Exception as e:
    if target == 'web_service' and _web_started:
        stop_web_service()
    raise
finally:
    # Keep container running for subsequent cells (will be stopped after parity)
    pass
```

**Cell 7 — Batch latency benchmark (Python code cell, PERF-03):**

```python
# PERF-03: Batch latency at 10, 100, 1000, 10000 rows
BATCH_SIZES = [10, 100, 1000, 10_000]
batch_results = {}

try:
    for n in BATCH_SIZES:
        if n > len(test_data):
            rows = pd.concat([test_data] * (n // len(test_data) + 1)).head(n).reset_index(drop=True)
        else:
            rows = test_data.head(n)

        t0 = time.perf_counter()
        for _, row in rows.iterrows():
            run_single_predict(row[feature_names].to_dict())
        elapsed = time.perf_counter() - t0
        batch_results[n] = {
            'total_s':     round(elapsed, 3),
            'mean_ms':     round(elapsed / n * 1000, 3),
            'rows_per_sec': round(n / elapsed, 1)
        }
        print(f"Batch {n:>6}: {batch_results[n]['mean_ms']:.3f} ms/row — {batch_results[n]['rows_per_sec']:.0f} rows/s")
except Exception as e:
    if target == 'web_service' and _web_started:
        stop_web_service()
    raise

batch_df = pd.DataFrame(batch_results).T.reset_index().rename(columns={'index': 'batch_size'})
display(batch_df)
```

**Cell 8 — Parity test (Python code cell, PERF-04):**

```python
# PERF-04: Parity test — deployed vs in-memory model, last 100 rows (D-06)
try:
    parity_rows = X_test  # already last 100 rows
    inmem_preds = model_pipeline.predict(parity_rows).tolist()

    deployed_preds = []
    for _, row in parity_rows.iterrows():
        deployed_preds.append(run_single_predict(row[feature_names].to_dict()))

    if problem_type == 'regression':
        ok = np.allclose(
            np.array(deployed_preds, dtype=float),
            np.array(inmem_preds, dtype=float),
            atol=1e-4
        )
    else:
        # classification, clustering — exact match
        ok = np.array_equal(
            np.array([str(p) for p in deployed_preds]),
            np.array([str(p) for p in inmem_preds])
        )

    if not ok:
        if problem_type == 'regression':
            diffs = np.abs(np.array(deployed_preds, dtype=float) - np.array(inmem_preds, dtype=float))
            n_diff = int(np.sum(diffs > 1e-4))
        else:
            n_diff = int(np.sum(
                np.array([str(p) for p in deployed_preds]) != np.array([str(p) for p in inmem_preds])
            ))
        raise AssertionError(
            f"PARITY FAILED: {n_diff} of {len(inmem_preds)} predictions differ "
            f"(tolerance: {'atol=1e-4' if problem_type == 'regression' else 'exact match'})"
        )

    print(f"PARITY PASSED: {len(inmem_preds)} predictions match "
          f"({'atol=1e-4' if problem_type == 'regression' else 'exact match'})")

finally:
    # Stop web service container after parity test
    if target == 'web_service' and _web_started:
        stop_web_service()
        _web_started = False
        print("Web service container stopped.")
```
